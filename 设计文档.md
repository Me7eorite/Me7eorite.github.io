实现：Quiet Outline 风格的本页目录 + 显隐切换按钮
参考来源
Obsidian 插件 Quiet Outline (https://github.com/guopenghui/obsidian-quiet-outline) 的核心特性：

默认折叠，点击箭头展开
不会因为滚动/点击自动展开
顶部一键切换显示层级（如「只看 h1」「展开到 h3」「全部展开」）
搜索框快速过滤
记住每篇文章的展开状态（用 sessionStorage）

功能需求
A. 替换 VitePress 默认 Outline

默认状态：所有 h1 折叠（只显示 h1 列表，h2/h3/h4 隐藏）
点击 h1 左侧的 ▶ 箭头展开/折叠该 h1 下的子项
滚动时高亮当前阅读位置对应的标题（不自动展开）

B. 右上角加一个显示/隐藏目录的按钮

在导航栏右上角加一个图标按钮（如 📑 或列表图标）
点击切换右侧 aside（包含 Outline）的显示/隐藏
状态用 localStorage 记忆（全局）
隐藏时主内容区域应该自动撑满变宽

2. 创建组件 .vitepress/theme/components/QuietOutline.vue
vue<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted, watch } from 'vue'
import { useData, useRoute } from 'vitepress'

interface Header {
  level: number
  title: string
  link: string
  children?: Header[]
}

const route = useRoute()
const { frontmatter, theme } = useData()

const headers = ref<Header[]>([])
const expandedKeys = ref<Set<string>>(new Set())
const activeLink = ref('')
const searchQuery = ref('')
const levelFilter = ref<'L2' | 'L3' | 'L4' | 'All'>('L2') // 默认只显示 h2

// 存储 key（按页面路径）
const storageKey = computed(() => `quiet-outline:${route.path}`)

// 收集页面 headers
function collectHeaders(): any[] {
  const main = document.querySelector('.VPDoc .vp-doc') || document.querySelector('.vp-doc')
  if (!main) return []
  const nodes = main.querySelectorAll('h2, h3, h4')
  const list: any[] = []
  nodes.forEach((el) => {
    const id = el.id
    if (!id) return
    list.push({
      level: parseInt(el.tagName.substring(1)),
      title: (el.textContent || '').replace(/#$/, '').trim(),
      link: '#' + id
    })
  })
  return list
}

// 扁平转树
function buildTree(flat: any[]): Header[] {
  const result: Header[] = []
  let curH2: Header | null = null
  let curH3: Header | null = null
  for (const h of flat) {
    const node: Header = { ...h, children: [] }
    if (h.level === 2) {
      curH2 = node; curH3 = null; result.push(node)
    } else if (h.level === 3 && curH2) {
      curH3 = node; curH2.children!.push(node)
    } else if (h.level === 4) {
      if (curH3) curH3.children!.push(node)
      else if (curH2) curH2.children!.push(node)
    }
  }
  return result
}

function refresh() {
  headers.value = buildTree(collectHeaders())
  // 恢复展开状态
  try {
    const saved = sessionStorage.getItem(storageKey.value)
    if (saved) {
      expandedKeys.value = new Set(JSON.parse(saved))
    } else {
      expandedKeys.value.clear()
    }
  } catch {}
}

function persist() {
  try {
    sessionStorage.setItem(storageKey.value, JSON.stringify([...expandedKeys.value]))
  } catch {}
}

function toggle(link: string) {
  if (expandedKeys.value.has(link)) expandedKeys.value.delete(link)
  else expandedKeys.value.add(link)
  expandedKeys.value = new Set(expandedKeys.value)
  persist()
}

function expandAll() {
  const all = new Set<string>()
  headers.value.forEach(h => {
    if (h.children?.length) all.add(h.link)
    h.children?.forEach(c => { if (c.children?.length) all.add(c.link) })
  })
  expandedKeys.value = all
  persist()
}

function collapseAll() {
  expandedKeys.value = new Set()
  persist()
}

// 层级过滤：决定是否渲染某个层级
function shouldShow(level: number): boolean {
  if (levelFilter.value === 'All') return true
  if (levelFilter.value === 'L2') return level <= 2
  if (levelFilter.value === 'L3') return level <= 3
  if (levelFilter.value === 'L4') return level <= 4
  return true
}

// 搜索过滤
function matchSearch(title: string): boolean {
  if (!searchQuery.value.trim()) return true
  return title.toLowerCase().includes(searchQuery.value.toLowerCase())
}

// 滚动高亮
function onScroll() {
  const headings = document.querySelectorAll('.VPDoc .vp-doc :is(h2, h3, h4)')
  let current = ''
  headings.forEach((h) => {
    const rect = h.getBoundingClientRect()
    if (rect.top < 120) current = '#' + h.id
  })
  activeLink.value = current
}

// 路由变化时刷新
watch(() => route.path, () => {
  setTimeout(refresh, 100)
})

onMounted(() => {
  setTimeout(refresh, 100)
  window.addEventListener('scroll', onScroll, { passive: true })
})
onUnmounted(() => {
  window.removeEventListener('scroll', onScroll)
})

const label = computed(() => {
  return frontmatter.value.outline?.label || theme.value.outline?.label || '本页目录'
})

// 过滤后的 headers
const filteredHeaders = computed(() => {
  if (!searchQuery.value.trim()) return headers.value
  // 搜索时把所有匹配项的父节点也展开
  const q = searchQuery.value.toLowerCase()
  return headers.value.filter(h => {
    const matchSelf = h.title.toLowerCase().includes(q)
    const matchChild = h.children?.some(c =>
      c.title.toLowerCase().includes(q) ||
      c.children?.some(cc => cc.title.toLowerCase().includes(q))
    )
    return matchSelf || matchChild
  })
})
</script>

<template>
  <div class="quiet-outline" v-if="headers.length > 0">
    <div class="qo-header">
      <div class="qo-title">{{ label }}</div>
      <div class="qo-toolbar">
        <input
          v-model="searchQuery"
          type="text"
          class="qo-search"
          placeholder="搜索标题..."
        />
        <div class="qo-levels">
          <button
            v-for="lv in ['L2', 'L3', 'L4', 'All']"
            :key="lv"
            class="qo-level-btn"
            :class="{ active: levelFilter === lv }"
            @click="levelFilter = lv as any"
          >{{ lv }}</button>
        </div>
        <div class="qo-actions">
          <button class="qo-action" @click="expandAll" title="全部展开">⊕</button>
          <button class="qo-action" @click="collapseAll" title="全部折叠">⊖</button>
        </div>
      </div>
    </div>

    <ul class="qo-list">
      <li
        v-for="h in filteredHeaders"
        :key="h.link"
        class="qo-item level-2"
      >
        <div class="qo-row" :class="{ active: activeLink === h.link }">
          <span
            v-if="h.children && h.children.length > 0 && shouldShow(3)"
            class="qo-caret"
            :class="{ expanded: expandedKeys.has(h.link) }"
            @click.stop.prevent="toggle(h.link)"
          >▶</span>
          <span v-else class="qo-caret-placeholder"></span>
          <a :href="h.link" class="qo-link">{{ h.title }}</a>
        </div>

        <ul
          v-if="h.children?.length && expandedKeys.has(h.link) && shouldShow(3)"
          class="qo-sublist"
        >
          <li
            v-for="c in h.children.filter(c => shouldShow(c.level) && matchSearch(c.title))"
            :key="c.link"
            :class="`qo-item level-${c.level}`"
          >
            <div class="qo-row" :class="{ active: activeLink === c.link }">
              <span
                v-if="c.children?.length && shouldShow(4)"
                class="qo-caret"
                :class="{ expanded: expandedKeys.has(c.link) }"
                @click.stop.prevent="toggle(c.link)"
              >▶</span>
              <span v-else class="qo-caret-placeholder"></span>
              <a :href="c.link" class="qo-link">{{ c.title }}</a>
            </div>
            <ul
              v-if="c.children?.length && expandedKeys.has(c.link) && shouldShow(4)"
              class="qo-sublist"
            >
              <li v-for="d in c.children" :key="d.link" class="qo-item level-4">
                <a :href="d.link" class="qo-link" :class="{ active: activeLink === d.link }">{{ d.title }}</a>
              </li>
            </ul>
          </li>
        </ul>
      </li>
    </ul>
  </div>
</template>

3. 创建显隐切换按钮组件 .vitepress/theme/components/OutlineToggle.vue
vue<script setup lang="ts">
import { ref, onMounted } from 'vue'

const visible = ref(true)

function toggle() {
  visible.value = !visible.value
  document.body.classList.toggle('outline-hidden', !visible.value)
  try {
    localStorage.setItem('outline-visible', String(visible.value))
  } catch {}
}

onMounted(() => {
  try {
    const saved = localStorage.getItem('outline-visible')
    if (saved === 'false') {
      visible.value = false
      document.body.classList.add('outline-hidden')
    }
  } catch {}
})
</script>

<template>
  <button class="outline-toggle-btn" @click="toggle" :title="visible ? '隐藏目录' : '显示目录'">
    <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
      <line x1="8" y1="6" x2="21" y2="6"/>
      <line x1="8" y1="12" x2="21" y2="12"/>
      <line x1="8" y1="18" x2="21" y2="18"/>
      <line x1="3" y1="6" x2="3.01" y2="6"/>
      <line x1="3" y1="12" x2="3.01" y2="12"/>
      <line x1="3" y1="18" x2="3.01" y2="18"/>
    </svg>
  </button>
</template>

右上角导航栏多出一个"列表"图标按钮，点击切换右侧目录显隐（状态全局记忆）
右侧目录顶部有：搜索框、L2/L3/L4/All 层级切换、全部展开/折叠按钮
